35 /*
36  * \brief setup default-storage tblobs from source NDArrays. If any source NDArray has non-default
37  * storage, it creates a temp NDArray with default storage and uses the temp tblob. The
38  * function also records the indices of non-default source NDArrays and the indices of
39  * their corresponding temporary NDArrays in the temp array.
40  * \param src list of source NDArray
41  * \param blobs list of tblobs to return
42  * \param temp_src list of source NDArrays which requires temporary default storage representation
43  * \param temp_dst list of temporary destination NDArrays for default storage representation
44  * \param idx_map mapping from indices in source NDArrays to indices in temp_dst. When not set,
45  indices are not recorded
46  * \return true if any source NDArray need to cast storage
47  */
48 inline bool SetupDefaultBlobsIn(const std::vector<NDArray>& src,
49  const std::vector<NDArray> *bufs,
50  std::vector<TBlob> *blobs,
51  std::vector<NDArray> *temp_src,
52  std::vector<NDArray> *temp_dst,
53  std::unordered_map<uint32_t, uint32_t> *idx_map) {
54  bool require_cast = false;
55  for (size_t i = 0; i < src.size(); i++) {
56  auto& nd = src[i];
57  bool is_default = nd.storage_type() == kDefaultStorage;
58 #if MXNET_USE_MKLDNN == 1
59  // We have to make sure it's default storage and default layout.
60  is_default = nd.IsDefaultData();
61 #endif
62  if (!is_default) {
63  (*idx_map)[i] = temp_dst->size();
64  NDArray temp = bufs != nullptr ? bufs->at(i) : NDArray(nd.shape(), nd.ctx(),
65  true, nd.dtype());
66 #if MXNET_USE_MKLDNN == 1
67  CHECK(temp.IsDefaultData());
68 #endif
69  temp_src->emplace_back(nd);
70  temp_dst->emplace_back(temp);
71  blobs->emplace_back(temp.data());
72  require_cast = true;
73  } else {
74  blobs->push_back(nd.data());
75  }
76  }
77  return require_cast;
78 }

80 inline bool SetupDefaultBlobsOut(const std::vector<NDArray>& src,
81  const std::vector<NDArray> *bufs,
82  std::vector<OpReqType> *req,
83  std::vector<TBlob> *blobs,
84  std::vector<NDArray> *temp_src,
85  std::vector<NDArray> *temp_dst) {
86  bool require_cast = false;
87  for (size_t i = 0; i < src.size(); i++) {
88  auto& nd = src[i];
89  bool is_default = nd.storage_type() == kDefaultStorage;
90 #if MXNET_USE_MKLDNN == 1
91  if (req->at(i) == kWriteInplace && nd.IsMKLDNNData())
92  // If it's write inplace and the output array doesn't use the default
93  // layout, we'll generate a temporary output array below, which means
94  // the input array and the output array are no longer the same array.
95  // we should change the request type.
96  req->at(i) = kWriteTo;
97  // We have to make sure it's default storage and default layout.
98  is_default = nd.IsDefaultData();
99 #endif
100  if (!is_default) {
101 #if MXNET_USE_MKLDNN == 1
102  NDArray temp;
103  if (bufs != nullptr) {
104  temp = bufs->at(i);
105  } else if (kAddTo == req->at(i) && nd.IsMKLDNNData()) {
106  temp = nd.Reorder2Default();
107  } else if (kAddTo == req->at(i)) {
108  temp = nd;
109  } else {
110  temp = NDArray(nd.shape(), nd.ctx(), true, nd.dtype());
111  }
112  CHECK(temp.IsDefaultData());
113 #else
114  NDArray temp = bufs != nullptr ? bufs->at(i) : NDArray(nd.shape(), nd.ctx(),
115  true, nd.dtype());
116 #endif
117  temp_src->emplace_back(nd);
118  temp_dst->emplace_back(temp);
119  blobs->emplace_back(temp.data());
120  require_cast = true;
121  } else {
122  blobs->push_back(nd.data());
123  }
124  }
125  return require_cast;
126 }

128 /*
129  * \brief setup default-storage tblobs for input and output NDArrays.
130  * If any NDArray has non-default storage,
131  * it creates a temp NDArray with default storage and uses the temp tblob. The
132  * function also records the indices of non-default source NDArrays and the indices of
133  * their corresponding temporary NDArrays in the temp array.
134  */
135 inline void SetupDefaultBlobsInOut(const std::vector<NDArray> &ndinputs,
136  const std::vector<NDArray> &ndoutputs,
137  const std::vector<NDArray> *in_bufs,
138  const std::vector<NDArray> *out_bufs,
139  std::vector<OpReqType> *req,
140  std::vector<TBlob> *input_blobs,
141  std::vector<TBlob> *output_blobs,
142  std::vector<NDArray> *pre_temp_src,
143  std::vector<NDArray> *pre_temp_dst,
144  std::vector<NDArray> *post_temp_src,
145  std::vector<NDArray> *post_temp_dst,
146  std::unordered_map<uint32_t, uint32_t> *in_temp_idx_map,
147  const std::vector<uint32_t> &mutate_idx) {
148  // populate input blobs
149  SetupDefaultBlobsIn(ndinputs, in_bufs, input_blobs, pre_temp_src, pre_temp_dst,
150  in_temp_idx_map);
151  // populate output blobs
152  SetupDefaultBlobsOut(ndoutputs, out_bufs, req, output_blobs, post_temp_dst,
153  post_temp_src);
154  // add mutable inputs to post temp list
155  for (const auto idx : mutate_idx) {
156  auto map_iter = in_temp_idx_map->find(idx);
157  if (map_iter != in_temp_idx_map->end()) {
158  post_temp_src->push_back(pre_temp_dst->at(map_iter->second));
159  post_temp_dst->push_back(ndinputs[idx]);
160  }
161  }
162 }

164 /*
165  * \brief cast the NDArrays in `src` and store the result in NDArrays in `dst`.
166  * This is only used for storage fallback in executor.
167  * \param src list of source NDArray to cast
168  * \param dst list of destionation NDArray which hold the result of cast_storage operation
169  * \param ctx operator context for cast_storage operation
170  */
171 inline void CastNonDefaultStorage(const std::vector<NDArray>& src,
172  const std::vector<NDArray>& dst,
173  const OpContext& ctx,
174  const bool is_gpu) {
175  CHECK_EQ(dst.size(), src.size());
176  for (size_t i = 0; i < src.size(); i++) {
177  if (is_gpu) {
178 #if MXNET_USE_CUDA
179  CastStorageDispatch<gpu>(ctx, src[i], dst[i]);
180 #else
181  LOG(FATAL) << MXNET_GPU_NOT_ENABLED_ERROR;
182 #endif
183  } else {
184  CastStorageDispatch<cpu>(ctx, src[i], dst[i]);
185  }
186  }

192 inline bool SameType(const nnvm::NodeAttrs& attrs,
193  std::vector<int> *iattr,
194  std::vector<int> *oattr) {
195  int def_v = -1;
196  for (int v : *oattr) {
197  if (v != -1) {
198  def_v = v; break;
199  }
200  }
201  if (def_v == -1) {
202  for (int v : *iattr) {
203  if (v != -1) {
204  def_v = v; break;
205  }
206  }
207  }
208  if (def_v == -1) return false;
209  for (int& v : *oattr) {
210  v = def_v;
211  }
212  for (int& v : *iattr) {
213  v = def_v;
214  }
215  return true;
216 }

224 inline bool DefaultStorageType(const nnvm::NodeAttrs& attrs,
225  const int dev_mask,
226  DispatchMode* dispatch_mode,
227  std::vector<int> *iattr,
228  std::vector<int> *oattr) {
229  bool fallback = false;
230  for (int& v : *oattr) {
231  if (v == -1) v = kDefaultStorage;
232  if (v != kDefaultStorage) fallback = true;
233  }
234  for (int& v : *iattr) {
235  if (v == -1) v = kDefaultStorage;
236  if (v != kDefaultStorage) fallback = true;
237  }
238  if (*dispatch_mode == DispatchMode::kUndefined) {
239  if (fallback) {
240  *dispatch_mode = DispatchMode::kFComputeFallback;
241  } else {
242  *dispatch_mode = DispatchMode::kFCompute;
243  }
244  }
245  return true;
246 }

281 inline void LogMemoryPlan(const nnvm::Graph& g) {
282  const auto &idx = g.indexed_graph();
283  const auto& vshape = g.GetAttr<nnvm::ShapeVector>("shape");
284  const auto& vtype = g.GetAttr<nnvm::DTypeVector>("dtype");
285  const auto& vstorage = g.GetAttr<nnvm::StorageVector>("storage_id");
286  // find node range
287  uint32_t node_start = 0, node_end = idx.num_nodes();
288  if (g.attrs.count("node_range")) {
289  const auto& range = g.GetAttr<std::pair<uint32_t, uint32_t> >("node_range");
290  node_start = range.first;
291  node_end = range.second;
292  }
293  for (uint32_t nid = node_start; nid < node_end; ++nid) {
294  const auto& inode = idx[nid];
295  if (inode.source->is_variable()) {
296  LOG(INFO) << "node " << nid << " var";
297  } else {
298  LOG(INFO) << "node " << nid << " " << inode.source->attrs.op->name;
299  for (const auto& e : inode.inputs) {
300  auto eid = idx.entry_id(e);
301  size_t kilo_bytes = vshape[eid].Size() * mshadow::mshadow_sizeof(vtype[eid]) / 1024;
302  LOG(INFO) << "\t\tinput " << eid << ": " << vshape[eid] << " ("
303  << kilo_bytes << " KB) -> " << storage_str(vstorage[eid]);
304  }
305  for (uint32_t index = 0; index < inode.source->num_outputs(); ++index) {
306  uint32_t eid = idx.entry_id(nid, index);
307  size_t kilo_bytes = vshape[eid].Size() * mshadow::mshadow_sizeof(vtype[eid]) / 1024;
308  LOG(INFO) << "\t\toutput " << eid << ": " << vshape[eid] << " ("
309  << kilo_bytes << " KB) -> " << storage_str(vstorage[eid]);
310  }
311  }
312  }
313 }

340 inline void LogInferStorage(const nnvm::Graph& g) {

343  const auto& dispatch_modes = g.GetAttr<DispatchModeVector>("dispatch_mode");
344  uint32_t node_start = 0, node_end = idx.num_nodes();
345  if (g.attrs.count("node_range")) {
346  const auto& range = g.GetAttr<std::pair<uint32_t, uint32_t> >("node_range");
347  node_start = range.first;

350  for (uint32_t nid = node_start; nid < node_end; ++nid) {
351  const auto& inode = idx[nid];
352  if (inode.source->is_variable()) {
353  LOG(INFO) << "node " << nid << " var";
354  } else {
355  LOG(INFO) << "node " << nid << " " << inode.source->attrs.op->name
356  << ": " << dispatch_mode_string(dispatch_modes[nid]);
357  for (const auto& e : inode.inputs) {
358  auto eid = idx.entry_id(e);
359  LOG(INFO) << "\t\tinput " << eid << ": " << stype_string(vstorage_type[eid]);
360  }
361  for (uint32_t index = 0; index < inode.source->num_outputs(); ++index) {
362  uint32_t eid = idx.entry_id(nid, index);
363  LOG(INFO) << "\t\toutput " << eid << ": " << stype_string(vstorage_type[eid]);
364  }
